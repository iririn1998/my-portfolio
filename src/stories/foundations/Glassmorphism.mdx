## Glass Background

export const GlassCard = ({ variable, label, dark }) => (
  <div
    style={{
      padding: "2rem",
      borderRadius: "16px",
      background: `var(${variable})`,
      backdropFilter: "var(--glass-blur-lg)",
      border: "var(--glass-border)",
      boxShadow: "var(--glass-shadow)",
      flex: 1,
    }}
  >
    <div style={{ fontWeight: 600, marginBottom: "0.5rem" }}>{label}</div>
    <code
      style={{
        fontSize: "0.75rem",
        color: dark ? "var(--color-neutral-400)" : "var(--color-neutral-300)",
      }}
    >
      {variable}
    </code>
  </div>
);

<div
  style={{

    gap: "1rem",
  }}
>
  <GlassCard variable="--glass-bg-light" label="Light (10% opacity)" />
  <GlassCard variable="--glass-bg-medium" label="Medium (15% opacity)" />
  <GlassCard variable="--glass-bg-strong" label="Strong (25% opacity)" />
